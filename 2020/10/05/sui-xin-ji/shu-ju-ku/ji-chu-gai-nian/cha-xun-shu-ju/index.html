select 查询语句

SELECT
    prod_name 
FROM
    products;
SELECT
    * 
FROM
    products;

distinct 数据库只返回不同的值

SELECT DISTINCT
    vend_id 
FROM
    products;

限制结果 limit 5 读取五条数据 offset 5 从第五条开始读取

SELECT
    prod_name 
FROM
    products 
    LIMIT 5 OFFSET 5;

排序数据 语句放在最后 order by 默认asc 升序排列 desc 降序排列

SELECT
    prod_name 
FROM
    products 
ORDER BY
    prod_name;

按多个排序

SELECT
    prod_id,
    prod_price,
    prod_name 
FROM
    products 
ORDER BY
    prod_price,
    prod_name;

过滤数据 where

SELECT
    prod_name,
    prod_price 
FROM
    products 
WHERE
    prod_price = 3.49;

范围值检查 between 开始值 and 结束值0

SELECT
    prod_name,
    prod_price 
FROM
    products 
WHERE
    prod_price BETWEEN 5 
    AND 10;

空值检查 where xx is null

SELECT
    prod_name 
FROM
    products 
WHERE
    prod_price IS NULL;
SELECT
    cust_name 
FROM
    customers 
WHERE
    cust_email IS NULL;

高级数据检查

组合where and

SELECT
    prod_id,
    prod_price,
    prod_name 
FROM
    products 
WHERE
    vend_id = 'DLL01' 
    AND prod_price <= 4; # 为避免条件组合错误 使用() 对操作符进行分组
SELECT prod_name, prod_price FROM products WHERE vend_id = 'DLL01' OR vend_id = "BRS01" AND prod_price >= 10;
SELECT
    prod_name,
    prod_price 
FROM
    products 
WHERE
    ( vend_id = 'DLL01' OR vend_id = "BRS01" ) 
    AND prod_price >= 10;

IN 操作符 指定条件范围

SELECT
    vend_id,
    prod_name,
    prod_price 
FROM
    products 
WHERE
    vend_id IN ( 'DLL01', 'BRS01' ) 
ORDER BY
    prod_name;

not where语句中用来否定其后条件的关键字

SELECT
    vend_id,
    prod_name 
FROM
    products 
WHERE
    NOT vend_id = 'DLL01' 
ORDER BY
    prod_name;

用通配符进行过滤

LIKE % 区分大小写

SELECT
    prod_id,
    prod_name 
FROM
    products 
WHERE
    prod_name LIKE 'Fish%';

    max( prod_price ) AS max_price

    products;

    min( prod_price ) AS max_price

分组数据

创建分组 GROUP BY

SELECT
    count( * ) AS num 
FROM
    products 
WHERE
    vend_id = 'DLL01';
SELECT
    vend_id,
    count( * ) AS num 
FROM
    products 
GROUP BY
    vend_id;

过滤分组 HAVING 在分组后进行过滤 where 在分组前对数据进行过滤

SELECT
    cust_id,
    count( * ) AS orders 
FROM
    orders 
GROUP BY
    cust_id 
HAVING
    count( * ) >= 2 SELECT
    vend_id,
    count( * ) AS num 
FROM
    products 
WHERE
    prod_price >= 4 
GROUP BY
    vend_id 
HAVING
    count( * ) >= 2 # 分组(group by)和排序(order by)
